% Captions: centred, no hanging indent under "Figure N", small font.
% Loading the caption package also disables KOMA's default hanging-indent caption.
\usepackage{caption}
\captionsetup{%
  labelformat=simple,      % "Figure 1.3" with no KOMA auto-dot (kills the "1.3.." double period)
  labelsep=period,         % -> "Figure 1.3. <caption>"
  labelfont=bf,
  font=small,
  justification=justified, % full text-width, aligned to the text margins (no hanging indent)
  skip=6pt}

% Floats: keep figures where they are written (fig-pos: H in _quarto.yml needs this),
% and, where a figure does float, let pages fill more fully and avoid float-only pages.
% Parts labelled by letter: "Part A", "Part B", ... (not "Part I").
\renewcommand{\thepart}{\Alph{part}}

\usepackage{float}
\setcounter{topnumber}{3}
\setcounter{bottomnumber}{2}
\setcounter{totalnumber}{5}
\renewcommand{\topfraction}{0.92}
\renewcommand{\bottomfraction}{0.85}
\renewcommand{\textfraction}{0.06}
\renewcommand{\floatpagefraction}{0.75}
